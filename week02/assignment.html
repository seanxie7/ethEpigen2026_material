3.2 Plot the distribution of the (spliced) length of protein-coding transcripts

filtered_exons <- exonsBy(ensdb_mouse, by = c("tx"), filter = ~ gene_biotype == "protein_coding")
spliced_lengths <- sum(width(filtered_exons))
hist(log10(spliced_lengths), 
     breaks = 100, 
     col = "olivedrab", 
     main = "Distribution of Spliced Lengths (Protein-Coding)",
     xlab = expression(log[10] * " Transcript Length"),
     ylab = "Frequency")
